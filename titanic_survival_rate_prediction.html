    '''

    ax[0].set_ylabel('Number of passengers')
    ax[0].set_title(feature+' VS Survival VS count')
    ax[0].legend(['dead','alive'])
    '''
    feature
    10
    feature
    '''
    train_data.groupby(feature)['Survived'].mean().plot(kind='bar',ax=ax[1])
    ax[1].set_ylabel('Survival Rate')

    plt.show()
